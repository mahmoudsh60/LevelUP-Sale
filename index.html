function doPost(e) { try { var doc = SpreadsheetApp.getActiveSpreadsheet(); // تأكد إن اسم الصفحة تحت في الإكسيل هو "الطلبات" var ordersSheet = doc.getSheetByName("الطلبات"); var data = JSON.parse(e.postData.contents); var date = new Date(); var formattedDate = date.toLocaleDateString('ar-EG') + ' - ' + date.toLocaleTimeString('ar-EG'); // الترتيب الجديد بناءً على صورتك: // A: التاريخ | B: الاسم | C: الموبايل | D: نوع التصريح | E: الالعاب | F: السعر الاجمالي | G: المساحه الاجماليه ordersSheet.appendRow([ formattedDate, // A data.name, // B data.phone, // C data.licenseType, // D (نوع التصريح المجمع) data.gamesFormatted, // E data.totalPrice, // F (السعر النهائي شامل الضريبة إن وجدت) data.totalSize // G ]); return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } catch(error) { return ContentService.createTextOutput("Error: " + error.toString()).setMimeType(ContentService.MimeType.TEXT); } }
